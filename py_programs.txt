  Recursive Factorial.py            def factorial(a):                                                                         
                                        if(a==1 or a==0):                                                                     
                                            return(1)                                                                         
                                        return(a*factorial(a-1))                                                              
                                                                                                                              
                                    x=int(input("Enter Number:- "))                                                           
                                    n=factorial(x)                                                                            
                                    print(n)                                                                                  
                                                                                                                                                                           
  Recursive Fibonacci.py            # Fibonacci Series Using Recursion                                                        
                                    def fibonacci(a):                                                                         
                                        if(a==0):                                                                             
                                            return(0)                                                                         
                                        if (a==1):                                                                            
                                            return(1)                                                                         
                                        return(fibonacci(a-1)+fibonacci(a-2))                                                 
                                                                                                                              
                                    x= int(input("Enter number:- "))                                                          
                                    n=fibonacci(x)                                                                            
                                    print(n)
